(* COMENTARIO *)

class Uno inherits IO {

    sum( init : Int, inil : Int) : Int {
        init - inil
    };

    count( initial : Int, ini : Int) : Int {
        initial + ini
    };

    contar( mas : Int, pas : Int) : Int {
        mas * pas
    };
    
};

class Main { 

   x : Uno <- new Uno;

   main() : Int {
        x @ Uno.contar(3,3)
 
    };

};
