(* COMENTARIO *)

class Uno inherits IO {

    x : Int;

    sum( init : Int, inil : Int) : Int {
        init - inil
    };

    count( initial : Int, ini : Int) : Int {
        initial + ini
    };

    contar() : Void {
        {
            x <- in_int();
            out_int(x);
        }
    };
    
};

class Main { 

   x : Uno <- new Uno;

   main() : Void {
        x @ Uno.contar()
 
    };

};
