(* COMENTARIO *)

class Uno inherits IO {
    
    count( initial : Int ) : Int {
        if initial < 10 then
            out_int(initial)
        else
            out_int(1)
        fi
    };

};

class Main { 

   x : Uno <- new Uno;

   main() : Int {
        x @ Uno.count(5)
        
    };

};
